\author{Anne Fayolle, Abhay Goel, Devlin Mallory, Eamon Quinlan-Gallego}

\begin{abstract}
We develop an algorithm for performing arithmetic operations in rings of finite length $p$-typical Witt vectors over finitely generated $\F_p$-algebras. We use this to find delta structures on quotients of polynomial rings. 
We them implement these algorithm as a package in the computer algebra system Macaulay2. 
\end{abstract}

\section{Finite length Witt vectors as finitely generated algebras}

\begin{proposition} \label{prop-witt-presentation-general}
	Let $\tilde R$ be a $p$-torsion free ring for which $R \ceq \tilde R / p \tilde R$ is reduced, and let $n \geq 1$ be an integer. Then the $(n-1)$-th ghost map induces an injective ring homomorphism
	%
	$$\begin{tikzcd}
		w_{n-1}' : W_n(R) \arrow[r, hookrightarrow] & \tilde R / p^n \tilde R
		\end{tikzcd}$$
	%
		whose image is $\tilde R^{p^{n-1}} + p \tilde R^{p^{n-2}} + \cdots + p^{n-1} \tilde R$.
\end{proposition}
%
\begin{proof}
	Since the $(n-1)$-th ghost map $w_{n-1}: W_n(\tilde R) \to \tilde R$ is given by $$w_{n-1}(x_0, \dots , x_{n-1}) = x_0^{p^{n-1}} + p x_1^{p^{n-2}} + p^{n-1} x_0,$$ we see that whenever $x_0, \dots , x_{n-1} \in p \tilde R$ we get $w_{n-1}(x_0, \dots , x_{n-1}) \in p^n \tilde R$. 
	\comment{Not sure how many details to give here: the easiest proof I see is that $p^x > x$ for every real number $x > 0$, and therefore $i + p^{n-1-i} > n$ for every $0 \leq i \leq n-1$} 
	This shows that the composition $W_n(\tilde R) \xrightarrow{w_{n-1}} \tilde R \to \tilde R / p^n \tilde R$ factors through $W_n(R)$ as claimed. The claim about the image is clear from the description of $w_{n-1}$ given above.

	It remains to show that $w_{n-1}'$ is injective; i.e., that given $x_0, \dots, x_n \in \tilde R$ we have $w_{n-1}(x_0, \dots , x_{n-1}) \in p^n \tilde R$ only if $x_0, \dots , x_n \in p \tilde R$. We prove this by induction on $n$, with the base case $n = 1$ being clear.

	Observe that $w_{n-1}(x_0, \dots, x_{n-1}) \equiv x_0^{p^{n-1}} \mod p \tilde R$, and from the assumption that $\tilde R / p \tilde R$ is reduced we obtain that $x_0 \in p \tilde R$, which in turn gives 
	%
	$$0 \equiv w_{n-1}(x_0, \dots , x_{n-1}) \equiv p w_{n-2} (x_1, \dots , x_{n-1}) \mod p^n \tilde R.$$
	%
	Using the $p$-torsion freeness of $\tilde R$, we conclude that $w_{n-2}(x_1, \dots , x_{n-1}) \in p^{n-1} \tilde R$, and now the claim follows from the induction hypothesis.
\end{proof}

\begin{corollary}
	The subset $\tilde R^{p^{n-1}} + p \tilde R^{p^{n-2}} + \cdots + p^{n-1} \tilde R$ is a subring of $\tilde R$.
\end{corollary}

\section{$\delta$-rings and $p$-derivations}
In this section, we explain how we can compute the set of possible $\delta$ structures on a given ring $A$.

It is well known that giving a $\delta$-structure on a ring $A$ is the same as giving a ring map $\phi\colon A\to W_2(A)$ such that $\pi\circ \phi=\id_A$ where $\pi\colon W_2(A)\to A$ is the projection onto the first coordinate. See e.g. \cite[Construction 2.1]{BhattPrismaticLectures2}.

Based on this, we are able to find equations that the $\delta$-structures on a given ring $A$ must solve. \comment{I don't think solve is the right word...}

If $A$ is a polynomial ring, say $A=\F_p[x_1,\ldots, x_n]$ then any map $\phi\colon A\to W_2(A)$ as above must send $f(x_1,\ldots, x_n)$ to $(f(x_1,\ldots, x_n),\delta(f(x_1,\ldots, x_n)))$. Note that this implies that $\delta(x_i)$ can be chosen arbitrarily for each i since 
\[
f((x_1,\delta(x_1)),\ldots, (x_n,\delta(x_n)))= (f(x_1,\ldots, x_n),\delta(f(x_1,\ldots, x_n))).
\]
Now, if $A$ is instead a quotient of a polynomial ring, say $A=k[x_1,\ldots,x_n]/(g_1,\ldots, g_r)$, then we again have that any map $\phi\colon A\to W_2(A)$ as above must send $f(x_1,\ldots, x_n)$ to $(f(x_1,\ldots, x_n),\delta(f(x_1,\ldots, x_n)))$ and that 
\[
f((x_1,\delta(x_1)),\ldots, (x_n,\delta(x_n)))= (f(x_1,\ldots, x_n),\delta(f(x_1,\ldots, x_n))).
\]
But in this case, this implies that the $(x_i,\delta(x_i))$ must satisfy the equations defined by the $f_j$s. Note that these are just equations in $W_2(A)$. So if we let $B\coloneqq A[y_1,\ldots,y_n]$ and consider the equations 
\begin{equation}\label{eqn delta is 0}
f_j((x_1,y_1),\ldots, (x_n,y_n))= (f_j(x_1,\ldots, x_n),\delta(f_j(x_1,\ldots, x_n)))=(0,0)
\end{equation}
then the solutions of the $y_j$s in terms of $x_i$s are precisely the possible values of the $\delta(x_i)$s.

If the field over which we are working isn't $\F_p$ but rather $\F_q$ for $q$ some power of $p$, we can still pretend we are working in over $\F_p$ by adding variables and having a bigger quotient so \autoref{eqn delta is 0} still holds. 

This is exactly what our \texttt{findFrobeniusLiftConstraints} method does. One can then use it to find examples of Frobenius lifts in different degrees using the \texttt{findFrobeniusLift} method. 

\begin{example} Here is what happens with the cusp for $p=2$.
    \begin{verbatim}
        i1: S=ZZ/2[x,y];
        i2: f=x^2-y^3;
        i3: findFrobeniusLiftConstraints (f)
                     3    2      2     4
        o3 = ideal (y  + x , aa x y + x )
                               1
                               
                       ZZ
        o3 : Ideal of  --[aa ..aa , x..y]
        
                        2   0    1

        i4 : findFrobeniusLift(f,2)

                        2	
        o4 = {x*y + y, y }

        o4 : List
        i5 : findFrobeniusLift(f,3)

               3    2       2    3    2                 3    2    2
        o5 = {x  + x y + x*y  + y  + x  + x*y + x + y, y  + x  + y }
    \end{verbatim}
\end{example}
This tells us that $aa_0=\delta(x)$ can be anything but $aa_1=\delta(y)$ must be such that $\delta(y)x^2y+x^4=0$. Replacing $x^2$ by $y^3$ gives $\delta(y)y^4+y^6=0$ so $\delta(y)\in y^2 +(f)$. This is exactly what \texttt{findFrobeniusLift(f,2)} returns. Trying for solutions with higher degrees will simply add elements of $(f)$ to the solution as one can see when trying \texttt{findFrobeniusLift(f,3)}.
\section{Quasi $F$-splittings}
We now explain how to use our code to detect whether or not a given variety is quasi $F$-split. %We first recall a Fedder-type criterion for quasi-$F$-splitting as in \cite[Theorem 5.8]{KawakamiTakamatsuYoshikawaFedderQuasi}. 
\begin{example} [\cite{KawakamiTakamatsuYoshikawaFedderQuasi}, Example 7.4] A Calabi-Yau surface of quasi $F$-splitting height $5$.
\begin{verbatim}
i1 : R=ZZ/3[x,y,z,w];

i2 : I=ideal(x^4 + y^4 + z^4 + w^4 + x^3*z + z^3*w + y*z^2*w + y*z*w^2)

            4    4    3     4      2     3         2    4
o2 = ideal(x  + y  + x z + z  + y*z w + z w + y*z*w  + w )

o2 : Ideal of R

i3 : quasiFSplittingNumber(I,1)

i4 : quasiFSplittingNumber(I,7)

o4 = 5
\end{verbatim}
\end{example}

The method \texttt{quasiFSplittingNumber} takes an ideal $I\subset R$ and a number $n$ and will return nothing in the case the quasi $F$-splitting number of $R/I$ is less than $n$, and the quasi $F$-splitting number of $R/I$ otherwise.
